89 lin_circle doublet_finding_helper::transform_coordinates(
90  const internal_spacepoint<spacepoint>& sp1,
91  const internal_spacepoint<spacepoint>& sp2) {
92 
93  static_assert(otherSpType == details::spacepoint_type::bottom ||
94  otherSpType == details::spacepoint_type::top);
95 
96  const scalar& xM = sp1.x();
97  const scalar& yM = sp1.y();
98  const scalar& zM = sp1.z();
99  const scalar& rM = sp1.radius();
100  const scalar& varianceZM = sp1.varianceZ();
101  const scalar& varianceRM = sp1.varianceR();
102  scalar cosPhiM = xM / rM;
103  scalar sinPhiM = yM / rM;
104 
105  scalar deltaX = sp2.x() - xM;
106  scalar deltaY = sp2.y() - yM;
107  scalar deltaZ = sp2.z() - zM;
108  // calculate projection fraction of spM->sp vector pointing in same
109  // direction as
110  // vector origin->spM (x) and projection fraction of spM->sp vector pointing
111  // orthogonal to origin->spM (y)
112  scalar x = deltaX * cosPhiM + deltaY * sinPhiM;
113  scalar y = deltaY * cosPhiM - deltaX * sinPhiM;
114  // 1/(length of M -> SP)
115  scalar iDeltaR2 =
116  static_cast<scalar>(1.) / (deltaX * deltaX + deltaY * deltaY);
117  scalar iDeltaR = std::sqrt(iDeltaR2);
118  // cot_theta = (deltaZ/deltaR)
119  scalar cot_theta = deltaZ * iDeltaR;
120  if constexpr (otherSpType == details::spacepoint_type::bottom) {
121  cot_theta = -cot_theta;
122  }
123  // VERY frequent (SP^3) access
124  lin_circle l;
125  l.m_cotTheta = cot_theta;
126  // location on z-axis of this SP-duplet
127  l.m_Zo = zM - rM * cot_theta;
128  l.m_iDeltaR = iDeltaR;
129  // transformation of circle equation (x,y) into linear equation (u,v)
130  // x^2 + y^2 - 2x_0*x - 2y_0*y = 0
131  // is transformed into
132  // 1 - 2x_0*u - 2y_0*v = 0
133  // using the following m_U and m_V
134  // (u = A + B*v); A and B are created later on
135  l.m_U = x * iDeltaR2;
136  l.m_V = y * iDeltaR2;
137  // error term for sp-pair without correlation of middle space point
138  l.m_Er = ((varianceZM + sp2.varianceZ()) +
139  (cot_theta * cot_theta) * (varianceRM + sp2.varianceR())) *
140  iDeltaR2;
141 
142  return l;
143 }
