22 inline std::pair<detray::axis::circular<>, detray::axis::regular<>> get_axes(
23  const spacepoint_grid_config& grid_config, vecmem::memory_resource& mr) {
24 
25  detray::dindex phiBins;
26  if (grid_config.bFieldInZ == 0) {
27  phiBins = 100;
28  } else {
29  // calculate circle intersections of helix and max detector radius
30  scalar minHelixRadius = grid_config.minPt / grid_config.bFieldInZ;
31 
32  // sanity check: if yOuter takes the square root of a negative number
33  if (minHelixRadius < grid_config.rMax / 2) {
34  throw std::domain_error(
35  "The value of minHelixRadius cannot be smaller than rMax / 2. "
36  "Please "
37  "check the configuration of bFieldInZ and minPt");
38  }
39  scalar maxR2 = grid_config.rMax * grid_config.rMax;
40  scalar xOuter = maxR2 / (2 * minHelixRadius);
41  scalar yOuter = std::sqrt(maxR2 - xOuter * xOuter);
42  scalar outerAngle = std::atan(xOuter / yOuter);
43 
44  // intersection of helix and max detector radius minus maximum R
45  // distance from middle SP to top SP
46  scalar innerAngle = 0;
47  scalar rMin = grid_config.rMax;
48  if (grid_config.rMax > grid_config.deltaRMax) {
49  rMin = grid_config.rMax - grid_config.deltaRMax;
50  scalar innerCircleR2 = (grid_config.rMax - grid_config.deltaRMax) *
51  (grid_config.rMax - grid_config.deltaRMax);
52  scalar xInner = innerCircleR2 / (2 * minHelixRadius);
53  scalar yInner = std::sqrt(innerCircleR2 - xInner * xInner);
54  innerAngle = std::atan(xInner / yInner);
55  }
56 
57  // evaluating the azimutal deflection including the maximum impact
58  // parameter
59  scalar deltaAngleWithMaxD0 =
60  std::abs(std::asin(grid_config.impactMax / (rMin)) -
61  std::asin(grid_config.impactMax / grid_config.rMax));
62 
63  // evaluating delta Phi based on the inner and outer angle, and the
64  // azimutal deflection including the maximum impact parameter Divide by
65  // config.phiBinDeflectionCoverage since we combine
66  // config.phiBinDeflectionCoverage number of consecutive phi bins in the
67  // seed making step. So each individual bin should cover
68  // 1/config.phiBinDeflectionCoverage of the maximum expected azimutal
69  // deflection
70  scalar deltaPhi = (outerAngle - innerAngle + deltaAngleWithMaxD0) /
71  grid_config.phiBinDeflectionCoverage;
72 
73  // sanity check: if the delta phi is equal to or less than zero, we'll
74  // be creating an infinite or a negative number of bins, which would be
75  // bad!
76  if (deltaPhi <= 0.) {
77  throw std::domain_error(
78  "Delta phi value is equal to or less than zero, leading to an "
79  "impossible number of bins (negative or infinite)");
80  }
81 
82  // divide 2pi by angle delta to get number of phi-bins
83  // size is always 2pi even for regions of interest
84  phiBins = std::ceil(2 * M_PI / deltaPhi);
85  // need to scale the number of phi bins accordingly to the number of
86  // consecutive phi bins in the seed making step.
87  // Each individual bin should be approximately a fraction (depending on
88  // this number) of the maximum expected azimutal deflection.
89  }
90 
91  detray::axis::circular m_phi_axis{phiBins, grid_config.phiMin,
92  grid_config.phiMax, mr};
93 
94  // TODO: can probably be optimized using smaller z bins
95  // and returning (multiple) neighbors only in one z-direction for forward
96  // seeds
97  // FIXME: zBinSize must include scattering
98 
99  scalar zBinSize = grid_config.cotThetaMax * grid_config.deltaRMax;
100  detray::dindex zBins = std::max(
101  1, (int)std::floor((grid_config.zMax - grid_config.zMin) / zBinSize));
102 
103  detray::axis::regular m_z_axis{zBins, grid_config.zMin, grid_config.zMax,
104  mr};
105 
106  return {m_phi_axis, m_z_axis};
107 }

112  return detray::detail::invalid_value<size_t>();

116  return detray::detail::invalid_value<size_t>();

122  return r_index;
