45 TRACCC_HOST inline void calc_cluster_properties(
46  const cell_collection_t& cluster, const cell_module& module, point2& mean,
47  point2& var, scalar& totalWeight) {
48 
49  // Loop over the cells of the cluster.
50  for (const cell& cell : cluster) {
51 
52  // Translate the cell readout value into a weight.
53  const scalar weight = signal_cell_modelling(cell.activation, module);
54 
55  // Only consider cells over a minimum threshold.
56  if (weight > module.threshold) {
57 
58  // Update all output properties with this cell.
59  totalWeight += cell.activation;
60  const point2 cell_position = position_from_cell(cell, module);
61  const point2 prev = mean;
62  const point2 diff = cell_position - prev;
63 
64  mean = prev + (weight / totalWeight) * diff;
65  for (std::size_t i = 0; i < 2; ++i) {
66  var[i] =
67  var[i] + weight * (diff[i]) * (cell_position[i] - mean[i]);
68  }
69  }
70  }

86 
87  // To calculate the mean and variance with high numerical stability

89  // single-pass online algorithm that works well for large numbers

92  // To learn more about this algorithm please refer to:
93  // [1] https://doi.org/10.1080/00401706.1962.10490022

95  // edition, chapter 4.2.2.
96 
97  // Calculate the cluster properties
98  scalar totalWeight = 0.;
99  point2 mean{0., 0.}, var{0., 0.};
100  detail::calc_cluster_properties(cluster, module, mean, var, totalWeight);
101 
102  if (totalWeight > 0.) {
103  measurement m;
104  m.module_link = module_link;
105  m.surface_link = module.surface_link;
106  // normalize the cell position
107  m.local = mean;
108  // normalize the variance
109  m.variance[0] = var[0] / totalWeight;
110  m.variance[1] = var[1] / totalWeight;
111  // plus pitch^2 / 12

115  pitch[1] * pitch[1] / static_cast<scalar>(12.)};
