46  // for errors on spM

48  static_cast<scalar>(2.f) *

50  spM.varianceZ()) *
51  lb.iDeltaR() * lt.iDeltaR();

55  scalar error;

59  // compare with scattering

61  deltaCotTheta = std::abs(deltaCotTheta);
62  // if deltaTheta larger than the scattering for the lower pT cut, skip
63  error = std::sqrt(error2);
64  dCotThetaMinusError2 = deltaCotTheta2 + error2 -
65  static_cast<scalar>(2.) * deltaCotTheta * error;

67  // if left side of ">" is positive, both sides of unequality can be
68  // squared

70  if (dCotThetaMinusError2 > scatteringInRegion2) {
71  return false;
72  }
73  }
