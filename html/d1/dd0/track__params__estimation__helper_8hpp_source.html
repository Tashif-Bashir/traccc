49  const auto& spT = sp_collection.at(seed.spT_link);
50 
51  darray<vector3, 3> sp_global_positions;
52  sp_global_positions[0] = spB.global;
53  sp_global_positions[1] = spM.global;
54  sp_global_positions[2] = spT.global;
55 
56  // Define a new coordinate frame with its origin at the bottom space
57  // point, z axis long the magnetic field direction and y axis
58  // perpendicular to vector from the bottom to middle space point.
59  // Hence, the projection of the middle space point on the tranverse
60  // plane will be located at the x axis of the new frame.
61  vector3 relVec = sp_global_positions[1] - sp_global_positions[0];
62  vector3 newZAxis = vector::normalize(bfield);
63  vector3 newYAxis = vector::normalize(vector::cross(newZAxis, relVec));
64  vector3 newXAxis = vector::cross(newYAxis, newZAxis);
65 
66  // The center of the new frame is at the bottom space point
67  vector3 translation = sp_global_positions[0];
68 
69  transform3 trans(translation, newZAxis, newXAxis);
70 
71  // The coordinate of the middle and top space point in the new frame
72  auto local1 = trans.point_to_local(sp_global_positions[1]);
73  auto local2 = trans.point_to_local(sp_global_positions[2]);
74 
75  // The uv1.y() should be zero
76  vector2 uv1 = uv_transform(local1[0], local1[1]);
77  vector2 uv2 = uv_transform(local2[0], local2[1]);
78 
79  // A,B are slope and intercept of the straight line in the u,v plane
80  // connecting the three points
81  scalar A = (uv2[1] - uv1[1]) / (uv2[0] - uv1[0]);
82  scalar B = uv2[1] - A * uv2[0];
83 
84  // Curvature (with a sign) estimate
85  scalar rho = -2.0f * B / getter::perp(vector2{1., A});
86  // The projection of the top space point on the transverse plane of
87  // the new frame
88  scalar rn = local2[0] * local2[0] + local2[1] * local2[1];
89  // The (1/tanTheta) of momentum in the new frame
90  static constexpr scalar G = static_cast<scalar>(1.f / 24.f);
91  scalar invTanTheta =
92  local2[2] * std::sqrt(1.f / rn) / (1.f + G * rho * rho * rn);
93 
94  // The momentum direction in the new frame (the center of the circle
95  // has the coordinate (-1.*A/(2*B), 1./(2*B)))
96  vector3 transDirection =
97  vector3({1., A, scalar(getter::perp(vector2{1., A})) * invTanTheta});
98  // Transform it back to the original frame
99  vector3 direction =
100  transform3::rotate(trans._data, vector::normalize(transDirection));
101 
102  // The estimated phi and theta
103  getter::element(params, e_bound_phi, 0) = getter::phi(direction);
104  getter::element(params, e_bound_theta, 0) = getter::theta(direction);
105 
106  // The measured loc0 and loc1
107  const auto& meas_for_spB = spB.meas;
108  getter::element(params, e_bound_loc0, 0) = meas_for_spB.local[0];
109  getter::element(params, e_bound_loc1, 0) = meas_for_spB.local[1];
110 
111  // The estimated q/pt in [GeV/c]^-1 (note that the pt is the
112  // projection of momentum on the transverse plane of the new frame)
113  scalar qOverPt = rho / getter::norm(bfield);
114  // The estimated q/p in [GeV/c]^-1
115  getter::element(params, e_bound_qoverp, 0) =
116  qOverPt / getter::perp(vector2{1., invTanTheta});
117 
118  // The estimated momentum, and its projection along the magnetic
119  // field diretion
120  scalar pInGeV = std::abs(1.0f / getter::element(params, e_bound_qoverp, 0));
121  scalar pzInGeV = 1.0f / std::abs(qOverPt) * invTanTheta;
122  scalar massInGeV = mass / unit<scalar>::GeV;
123 
124  // The estimated velocity, and its projection along the magnetic
125  // field diretion
126  scalar v = pInGeV / getter::perp(vector2{pInGeV, massInGeV});
127  scalar vz = pzInGeV / getter::perp(vector2{pInGeV, massInGeV});
128  // The z coordinate of the bottom space point along the magnetic
129  // field direction
130  scalar pathz =
131  vector::dot(sp_global_positions[0], bfield) / getter::norm(bfield);
132 
133  // The estimated time (use path length along magnetic field only if
134  // it's not zero)
135  if (pathz != 0) {
136  getter::element(params, e_bound_time, 0) = pathz / vz;
137  } else {
138  getter::element(params, e_bound_time, 0) =
139  getter::norm(sp_global_positions[0]) / v;
140  }

142  return params;
143 }
