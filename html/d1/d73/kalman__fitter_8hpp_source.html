200  TRACCC_HOST_DEVICE
201  void smooth(state& fitter_state) {
202  auto& track_states = fitter_state.m_fit_actor_state.m_track_states;
203 
204  // The smoothing algorithm requires the following:
205  // (1) the filtered track parameter of the current surface
206  // (2) the smoothed track parameter of the next surface
207  //
208  // Since the smoothed track parameter of the last surface can be
209  // considered to be the filtered one, we can reversly iterate the
210  // algorithm to obtain the smoothed parameter of other surfaces
211  auto& last = track_states.back();
212  last.smoothed().set_vector(last.filtered().vector());
213  last.smoothed().set_covariance(last.filtered().covariance());
214  last.smoothed_chi2() = last.filtered_chi2();
215 
216  for (typename vector_type<
217  track_state<transform3_type>>::reverse_iterator it =
218  track_states.rbegin() + 1;
219  it != track_states.rend(); ++it) {
220 
221  // Run kalman smoother
222  const detray::surface<detector_type> sf{m_detector,
223  it->surface_link()};
224  sf.template visit_mask<gain_matrix_smoother<transform3_type>>(
225  *it, *(it - 1));
226  }
227  }

231  auto& fit_info = fitter_state.m_fit_info;
232  auto& track_states = fitter_state.m_fit_actor_state.m_track_states;
233 
234  // Fit parameter = smoothed track parameter at the first surface
235  fit_info.fit_params = track_states[0].smoothed();
236 
237  for (const auto& trk_state : track_states) {
238 
239  const detray::surface<detector_type> sf{m_detector,
240  trk_state.surface_link()};
241  sf.template visit_mask<statistics_updater<transform3_type>>(
242  fit_info, trk_state);
243  }

246  fit_info.ndf = fit_info.ndf - 5.f;
