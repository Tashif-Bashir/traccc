29 container_t read_binary_container(std::string_view filename,
30  vecmem::memory_resource* mr = nullptr) {
31 
32  // Make sure that the chosen types work.
33  static_assert(std::is_standard_layout_v<typename container_t::header_type>,
34  "Container header type must be standard layout.");
35  static_assert(std::is_standard_layout_v<typename container_t::item_type>,
36  "Container item type must be standard layout.");
37 
38  // Open the input file.
39  std::ifstream in_file(filename.data(), std::ios::binary);
40 
41  // Read the size of the header vector.
42  std::size_t headers_size;
43  in_file.read(reinterpret_cast<char*>(&headers_size), sizeof(std::size_t));
44 
45  // Read the sizes of the item vector.
46  std::vector<std::size_t> items_size(headers_size);
47  in_file.read(reinterpret_cast<char*>(items_size.data()),
48  headers_size * sizeof(typename std::size_t));
49 
50  // Create the result container, and set it to the correct (outer) size right
51  // away.
52  container_t result(headers_size, mr);
53 
54  // Read the header payload into memory.
55  in_file.read(reinterpret_cast<char*>(result.get_headers().data()),
56  headers_size * sizeof(typename container_t::header_type));
57 
58  // Read the items in multiple steps.
59  for (std::size_t i = 0; i < headers_size; ++i) {
60  result.get_items().at(i).resize(items_size.at(i));
61  in_file.read(
62  reinterpret_cast<char*>(result.get_items().at(i).data()),
63  items_size.at(i) * sizeof(typename container_t::item_type));
64  }
65 
66  // Return the newly created container.
67  return result;
68 }
