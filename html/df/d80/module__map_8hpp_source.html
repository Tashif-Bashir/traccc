159  void create_tree(const std::vector<module_map_node>& nodes) {
160  /*
161  * First, we resize the number of nodes in the tree to the number of
162  * nodes in the input, but rounded up such that we have we end up with
163  * 2^n - 1 spaces, where n is the ceiling of log2 of the size of the
164  * input, plus one.
165  */
166  m_nodes.resize(round_up(nodes.size()));
167 
168  /*
169  * This vector is our work queue for the pre-order insertion that we
170  * will be doing. Each of these values contains a position where to
171  * insert the next node, and the subset of nodes to insert in that
172  * sub-tree.
173  */
174  std::vector<std::tuple<std::size_t, std::size_t, std::size_t>> z;
175 
176  /*
177  * To kick-start the algorithm, we want to insert all nodes (from 0 to
178  * the last one) at the first spot in the array.
179  */
180  z.emplace_back(0, 0, nodes.size());
181 
182  /*
183  * We are done when the work queue is empty, but we will keep adding
184  * things to it for a while at the start.
185  */
186  while (!z.empty()) {
187  /*
188  * Pop the current index and node range off the stack.
189  */
190  std::size_t i, s, e;
191  std::tie(i, s, e) = z.back();
192  z.pop_back();
193 
194  /*
195  * Pick a node roughly in the middle of the node range. This will
196  * be our next child.
197  */
198  std::size_t m = (s + e) / 2;
199 
200  /*
201  * Insert the node we selected as our center in the given position.
202  */
203  m_nodes[i] = nodes[m];
204 
205  /*
206  * If we have any nodes on the left side, we will put an element in
207  * the work queue to place the left nodes at index 2i + 1.
208  */
209  if (m - s > 0) {
210  z.emplace_back(2 * i + 1, s, m);
211  }
212 
213  /*
214  * Same thing for the right side, but we insert the right side
215  * starting at 2i + 2 instead.
216  */
217  if (e - (m + 1) > 0) {
218  z.emplace_back(2 * i + 2, m + 1, e);
219  }
220  }
221  }

264  static std::pair<std::vector<V>, std::vector<module_map_node>> node_list(
265  const std::map<K, V>& input) {
266  /*
267  * We start out by grabbing a list of all the keys in the map.
268  */
269  std::vector<K> keys;
270  keys.reserve(input.size());
271  for (const std::pair<const K, V>& i : input) {
272  keys.push_back(i.first);
273  }
274 
275  /*
276  * Next, we declare a vector of values, which is one of the two things
277  * that we will return in the end.
278  */
279  std::vector<V> values;
280  values.reserve(keys.size());
281 
282  /*
283  * To find contiguous sub-ranges, the keys must be sorted. In general,
284  * it is likely that they will be pre-sorted.
285  */
286  std::sort(keys.begin(), keys.end());
287 
288  /*
289  * This vector will hold our nodes. We'll update it as we go, and we
290  * start out with an incomplete first node, which represents the single
291  * value at the start of the key vector. We will add to this later!
292  */
293  std::vector<module_map_node> nodes;
294  nodes.push_back(module_map_node(keys.front(), 1, values.size()));
295  values.push_back(input.at(keys.front()));
296 
297  /*
298  * Starting from the second key, we check if they are contiuous with
299  * the last key, and store them if so.
300  */
301  for (std::size_t i = 1; i < keys.size(); ++i) {
302  /*
303  * If the key is not adjacent to the last one, we push a new
304  * partial node onto the node vector.
305  */
306  if (keys.at(i) != keys.at(i - 1) + 1) {
307  nodes.push_back(module_map_node(keys.at(i), 0, values.size()));
308  }
309 
310  /*
311  * Push the current value into the value array, and register one
312  * additional element in the current node.
313  */
314  ++nodes.back().size;
315  values.push_back(input.at(keys.at(i)));
316  }
317 
318  /*
319  * Return both the generated values.
320  */
321  return {values, nodes};
322  }

334  const V* at_helper(const K& i, std::size_t n) const {
335  /*
336  * For memory safety, if we are out of bounds we will exit.
337  */
338  if (n >= m_nodes.size()) {
339  return nullptr;
340  }
341 
342  /*
343  * Retrieve the current root node.
344  */
345  const module_map_node& node = m_nodes[n];
346 
347  /*
348  * If the size is zero, it is essentially an invalid node (i.e. the
349  * node does not exist).
350  */
351  if (node.size == 0) {
352  return nullptr;
353  }
354 
355  /*
356  * If the value we are looking for is past the start of the current
357  * node, there are three possibilities. Firstly, the value might be in
358  * the current node. Secondly, the value might be in the right child of
359  * the current node. Thirdly, the value might not be in the map at all.
360  */
361  if (i >= node.start) {
362  /*
363  * Next, we check if the value is within the range represented by
364  * the current node.
365  */
366  if (i < node.start + node.size) {
367  /*
368  * Found it! Return a pointer to the value within the
369  * contiguous range.
370  */
371  return &m_values[node.index + (i - node.start)];
372  } else {
373  /*
374  * Two possibilties remain, we need to check the right subtree.
375  */
376  return at_helper(i, 2 * n + 2);
377  }
378  }
379  /*
380  * If the value we want to find is less then the start of this node,
381  * there are only two possibilities. Firstly, the value might be in the
382  * left subtree, or the value might not be in the map at all.
383  */
384  else {
385  return at_helper(i, 2 * n + 1);
386  }
387  }
